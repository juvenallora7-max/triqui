<button onclick="resetGame()">Reiniciar</button>

<script>
let currentPlayer = "X";
let cells = [];
let gameActive = true;

const winningCombos = [
  [0,1,2],[3,4,5],[6,7,8],
  [0,3,6],[1,4,7],[2,5,8],
  [0,4,8],[2,4,6]
];

const board = document.getElementById("board");

for (let i = 0; i < 9; i++) {
  let button = document.createElement("button");
  button.classList.add("cell");
  button.addEventListener("click", () => handleClick(i));
  board.appendChild(button);
  cells.push(button);
}

function handleClick(index) {
  if (!gameActive || cells[index].textContent !== "") return;

  cells[index].textContent = currentPlayer;

  if (checkWinner()) {
    document.getElementById("status").textContent = "Ganó " + currentPlayer + " 🎉";
    gameActive = false;
    return;
  }

  if (cells.every(cell => cell.textContent !== "")) {
    document.getElementById("status").textContent = "Empate 🤝";
    gameActive = false;
    return;
  }

  currentPlayer = currentPlayer === "X" ? "O" : "X";
  document.getElementById("status").textContent = "Turno: " + currentPlayer;
}

function checkWinner() {
  return winningCombos.some(combo => {
    return combo.every(i => cells[i].textContent === currentPlayer);
  });
}

function resetGame() {
  cells.forEach(cell => cell.textContent = "");
  currentPlayer = "X";
  gameActive = true;
  document.getElementById("status").textContent = "Turno: X";
}
</script>
